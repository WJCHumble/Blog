## x 的平方根

```javascript
/**
 * @param {number} x
 * @return {number}
 */
var mySqrt = function(x) {
  if (x < 2) return x

  let l = 1, r = x, res;
  while(l <= r) {
    let mid = Math.floor((r + l) / 2)
    if (mid === x / mid)  {
      return mid
    }

    if (mid > x / mid) {
      r = mid - 1
    }

    if (mid < x / mid) {
      l = mid + 1
      res = mid
    }
  }

  return res
};
```

```javascript
/**
 * @param {number} x
 * @return {number}
 */
var mySqrt = function(x) {
  let r = x
  while(r * r > x) {
    r = Math.floor((r + x / r) / 2)
  }

  return r
}
```
